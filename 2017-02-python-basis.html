实例

统计一个TXT(SNP)文件中不同突变类型的个数

file = open("example.txt")

counts = {}

for n, line in enumerate(file):
    if n == 0:
        continue

    line = line.split()
    mut = line[3] + line[4]

    if mut in counts:
        counts[mut] = counts[mut] + 1
    else:
        counts[mut] = 1

print(counts)
